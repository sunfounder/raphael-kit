                time.sleep(3)
                i += 1
            else:
                print("waiting")
                time.sleep(0.5)

    # KEYBOARD INTERRUPT
    # ----------------------------
    except KeyboardInterrupt:
        print("\nKeyboard interrupt received. Exiting program...")

    # ----------------------------
    # CLEANUP
    # ----------------------------

        except:
