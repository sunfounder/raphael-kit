    int read_word_2c(int addr)
    {
        int val;
        val = wiringPiI2CReadReg8(fd, addr);
        val = val << 8;
        val += wiringPiI2CReadReg8(fd, addr+1);
        if (val >= 0x8000)

    double dist(double a, double b)
    {

    }

    double get_y_rotation(double x, double y, double z)
