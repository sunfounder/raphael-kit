    {
        pinMode(Gpin, OUTPUT);
        pinMode(Rpin, OUTPUT);

            digitalWrite(Rpin, HIGH);
            digitalWrite(Gpin, LOW);
        }

            digitalWrite(Rpin, LOW);
            digitalWrite(Gpin, HIGH);
        }

                temp = 1;
            }
            if ( digitalRead(speedPin) == 1 ){
                temp = 0;
            }

            LED(temp);
            Print(temp);
