    import spidev
    import time
    from gpiozero import PWMLED

    # Inicializar un LED PWM en el pin GPIO 22
    led = PWMLED(22)

    # Inicializar la comunicaci√≥n SPI (Bus 0, CE0 -> GPIO8)
    spi = spidev.SpiDev()
    spi.open(0, 0)  # Bus 0, CS0
    spi.max_speed_hz = 1000000  # 1 MHz

    # Funci√≥n para leer del canal MCP3008 (0‚Äì7)
    def read_adc(channel):
        """
        Leer el valor anal√≥gico del MCP3008 (0‚Äì1023)
        """
        if channel < 0 or channel > 7:
            return -1
        # Protocolo MCP3008: bit de inicio, modo unipolar, canal (3 bits), relleno
        r = spi.xfer2([1, (8 + channel) << 4, 0])
        value = ((r[1] & 3) << 8) | r[2]
        return value

    # Funci√≥n de mapeo de valores entre rangos
    def MAP(x, in_min, in_max, out_min, out_max):
        return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min

    # Bucle principal para leer el valor del ADC y controlar la intensidad del LED
    def loop():

            # Leer el valor anal√≥gico del canal 0 del MCP3008
            analogVal = read_adc(0)
            print('value = %d' % analogVal)

            # Mapear 0‚Äì1023 al rango PWM 0.0‚Äì1.0
            led.value = analogVal / 1023.0

            # Esperar 0.2 segundos
            time.sleep(0.2)

    # Ejecutar el bucle principal y manejar KeyboardInterrupt para un apagado seguro
    try:
        loop()
    except KeyboardInterrupt:
