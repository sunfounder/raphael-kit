        """

        :param channel: ADC-Kanal (0-7)
        :return: 10-Bit-Integer (0-1023)
        """

        value = ((adc[1] & 3) << 8) | adc[2]
        return value

        Wertebereich abbilden

        return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min
