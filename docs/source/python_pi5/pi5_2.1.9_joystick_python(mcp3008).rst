    import spidev
    import time

        """
        Lee el valor anal√≥gico del canal especificado del MCP3008 (0‚Äì7)
        :param channel: n√∫mero de canal ADC (0‚Äì7)
        :return: valor entero de 10 bits (0‚Äì1023)
